import pandas as pd

# 1Ô∏è‚É£ Baca file hasil filter sebelumnya
df = pd.read_csv("filtered_get_requests.csv")

# 2Ô∏è‚É£ Pastikan kolomnya benar
print("=== Nama Kolom ===")
print(df.columns)

# 3Ô∏è‚É£ Pastikan kolom 'Request time' ada
if 'Request time' not in df.columns:
    raise KeyError("Kolom 'Request time' tidak ditemukan. Cek nama kolom pada df.columns")

# 4Ô∏è‚É£ Konversi kolom 'Request time' ke format datetime
df['Request time'] = pd.to_datetime(df['Request time'], errors='coerce')

# 5Ô∏è‚É£ Ubah tampilannya jadi string format 'YYYY-MM-DD HH:MM:SS'
df['Request time'] = df['Request time'].dt.strftime('%Y-%m-%d %H:%M:%S')

# 6Ô∏è‚É£ Tampilkan hasil
print("\n=== 10 Baris Pertama Setelah Konversi ===")
print(df[['Request time']].head(10))

# 7Ô∏è‚É£ Simpan ke file baru
df.to_csv("converted_request_time.csv", index=False)
print("\n‚úÖ Kolom 'Request time' telah dikonversi dan disimpan sebagai 'converted_request_time.csv'")

import pandas as pd

# 1Ô∏è‚É£ Baca file hasil sebelumnya
df = pd.read_csv("converted_request_time.csv")

# 2Ô∏è‚É£ Tampilkan nama kolom untuk memastikan struktur
print("=== Nama Kolom ===")
print(df.columns)

# 3Ô∏è‚É£ Pastikan kolom yang dibutuhkan ada
required_cols = ['Remote host', 'Request Protocol']
for col in required_cols:
    if col not in df.columns:
        raise KeyError(f"Kolom '{col}' tidak ditemukan. Cek nama kolom pada df.columns")

# 4Ô∏è‚É£ (Opsional) Tampilkan daftar unik untuk membantu memilih filter
print("\n=== Contoh Nilai Unik ===")
print("Remote host unik:", df['Remote host'].unique()[:5])
print("Request Protocol unik:", df['Request Protocol'].unique())

# 5Ô∏è‚É£ Tentukan nilai filter (bisa diganti sesuai kebutuhan)
target_ip = '66.249.139.233'         # üß© contoh IP
target_protocol = 'HTTP/1.1'         # üß© contoh protokol

# 6Ô∏è‚É£ Lakukan filter
filtered_df = df[
    (df['Remote host'] == target_ip) &
    (df['Request Protocol'] == target_protocol)
]

# 7Ô∏è‚É£ Tampilkan hasil
print(f"\n=== Hasil Filter untuk IP {target_ip} & Protocol {target_protocol} ===")
print(filtered_df.head(10))

# 8Ô∏è‚É£ Hitung jumlah total baris hasil filter
print("\nJumlah total hasil filter:", len(filtered_df))

# 9Ô∏è‚É£ Simpan hasil ke file baru
filtered_df.to_csv("filtered_ip_protocol.csv", index=False)
print("\n‚úÖ Hasil filter disimpan sebagai 'filtered_ip_protocol.csv'")

import pandas as pd

# DIUBAH: Nama file input yang benar
file_name = "method_get_only_NASA.csv"
# DIUBAH: Nama kolom target yang benar
target_column = "time"

# 1. Muat dataset
print(f"Memuat data dari {file_name}...")
# DIUBAH: Tambahkan sep=';' untuk membaca file dengan benar
df = pd.read_csv(file_name, sep=';')

# 2. Periksa apakah kolom target ada
if target_column not in df.columns:
    print(f"Error: Kolom '{target_column}' tidak ditemukan di CSV.")
else:
    # 3. Tampilkan contoh format asli (ini adalah Angka/Timestamp)
    print(f"Format asli (baris pertama): {df[target_column].iloc[0]}")

    # 4. Lakukan transformasi:
    #    Kita tidak mengganti 'T' atau 'Z' karena data aslinya adalah angka (Unix timestamp).
    #    Kita akan MENGONVERSI timestamp itu ke format datetime yang mudah dibaca.
    print(f"Mengonversi kolom '{target_column}' (Unix timestamp) ke format Datetime...")

    # pd.to_datetime mengonversi angka timestamp (unit='s' berarti 'seconds')
    # Kita buat kolom baru 'datetime_readable' agar data asli tidak hilang
    df['datetime_readable'] = pd.to_datetime(df[target_column], unit='s')

    # 5. Simpan data yang telah diubah ke file CSV baru
    output_filename = "time_converted_NASA.csv"
    # DIUBAH: Tambahkan sep=';' agar konsisten
    df.to_csv(output_filename, index=False, sep=';')

    # 6. Cetak pesan sukses dan pratinjau
    print(f"\nBerhasil mengonversi data waktu dan menyimpannya ke {output_filename}.")
    print("Pratinjau data yang telah diubah (lihat kolom baru 'datetime_readable'):")
    print(df.head())

import pandas as pd

# 1. Tentukan file input (file yang memiliki 'datetime_readable')
file_name = "time_converted_NASA.csv"

# 2. Tentukan kolom-kolom yang akan kita gunakan
col_ip = "host"               # Kolom untuk mengelompokkan sesi
col_time = "datetime_readable" # Kolom untuk mengurutkan sesi
new_col_name = "SESSION"      # Nama kolom sesi baru

print(f"Memuat data dari {file_name}...")
try:
    # 3. Muat dataset
    # PENTING: Kita gunakan parse_dates=[col_time]
    # agar pandas langsung membacanya sebagai objek datetime
    df = pd.read_csv(file_name, sep=';', parse_dates=[col_time])
except FileNotFoundError:
    print(f"Error: File {file_name} tidak ditemukan.")
    print("Pastikan Anda sudah menjalankan skrip konversi 'time' sebelumnya.")
    exit()
except ValueError as e:
    print(f"Error saat membaca {file_name}. Cek kolom '{col_time}'.")
    print(f"Detail: {e}")
    exit()


# 4. Periksa apakah kolom yang diperlukan ada
if col_ip in df.columns and col_time in df.columns:

    # 5. BUAT KOLOM SESSION (Logika yang sama seperti sebelumnya)
    print(f"Membuat kolom '{new_col_name}' berdasarkan '{col_ip}' unik...")
    df[new_col_name] = df[col_ip].astype('category').cat.codes + 1

    # 6. Urutkan data berdasarkan SESSION, lalu WAKTU
    print(f"Mengurutkan data berdasarkan '{new_col_name}' dan '{col_time}'...")
    df_sorted = df.sort_values(by=[new_col_name, col_time], ascending=True)

    # 7. Hapus kolom 'time' (timestamp angka) yang asli agar rapi
    if 'time' in df_sorted.columns:
        print("Menghapus kolom 'time' (timestamp unix) asli...")
        df_sorted = df_sorted.drop(columns=['time'])

    # 8. Simpan DataFrame LENGKAP yang sudah diurutkan
    output_filename = "session_data_full_NASA.csv"

    # --- INI YANG DIUBAH ---
    # Kita ubah sep=';' (titik-koma) menjadi sep=',' (koma)
    print(f"Menyimpan data ke {output_filename} dengan pemisah KOMA (',')...")
    df_sorted.to_csv(output_filename, index=False, sep=',')
    # -------------------------

    print(f"\nBerhasil! Kolom '{new_col_name}' telah ditambahkan.")
    print(f"File baru disimpan sebagai: {output_filename}")
    print("\nPratinjau data baru (menampilkan kolom yang Anda minta):")

    # Tampilkan pratinjau hanya dengan kolom yang diminta agar mudah dilihat
    preview_cols = [new_col_name, col_ip, col_time, 'url']
    print(df_sorted[preview_cols].head())

else:
    print(f"Error: Kolom '{col_ip}' atau '{col_time}' tidak ditemukan di {file_name}.")
